\begin{frame}{Method}
  \gfx{lime_distance_weight}{.5}

  \begin{itemize}
    \item Sample points around $x_i$
    \item Use complex model to predict labels for each sample
    \item Weigh samples according to distance to $x_i$
        \item Learn new simple model on weighted samples
        \item Use simple model to explain
  \end{itemize}

\end{frame}
