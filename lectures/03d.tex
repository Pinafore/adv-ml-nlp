\begin{frame}{Adding Context Vectors}

\begin{itemize}

  \item \dots and context vectors $c$

\begin{frame}{Smoothing}

  \begin{itemize}
      \item Introduced in word2vec for negative sampling ($\alpha=0.75$)
      \begin{equation}
        \hat P_\alpha(c) = \frac{ \#(c)^\alpha}{\sum_{c'} \#(c)^\alpha}
      \end{equation}
    \item For PMI, helps remove bias toward rare words
    \pause
    \item And makes it about as good as word2vec
\end{itemize}
\end{frame}
